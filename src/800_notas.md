# Principios de d3

~~~~{javscript}
d3.select("body")              // select the body element
  .append("svg")               // add new svg element
    .attr("width", 250)        //  with some properties
    .attr("height", 50)
    .append("text")            // add a text element to svg element
      .text("Easy peasy")
      .attr("y", 25)
      .attr("x", 0)
      .style("fill", "purple")
~~~~
